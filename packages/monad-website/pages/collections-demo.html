How It Works

// json
{
  "testimonials": [
    {
      "name": "Sarah Johnson",
      "company": "TechCorp",
      "quote": "Monad is amazing!",
      "rating": 5
    }
  ]
}

// In your HTML template:
<!-- monad:loop testimonial in testimonials -->
  <div class="testimonial">
    <h3>{{testimonial.name}}</h3>
    <p>"{{testimonial.quote}}"</p>
    <span>Rating: {{testimonial.rating}}/5</span>
  </div>
<!-- monad:endloop -->
